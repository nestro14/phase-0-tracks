
CREATE TABLE person
  (
    person_id INTEGER PRIMARY KEY,
    first_name VARCHAR(255),
    last_name VARCHAR(255),
    age INT
  );

  CREATE TABLE alias
  (
    alias_id INTEGER PRIMARY KEY,
    alias VARCHAR(255),
    person_id INT,
    FOREIGN KEY (person_id)
    REFERENCES person(person_id)
  );
